/**
 * @param {number[]} target
 * @param {number[]} arr
 * @return {number}
 */
var minOperations = function(target, arr) {
    let temp = []
    for (let i = 0; i < arr.length; i++) {
        let curr = target.indexOf(arr[i])
        if (curr > -1) {
            if (target.indexOf(temp[temp.length - 1]) < curr || temp.length === 0) {
                temp.push(arr[i])
            } else {
                for (let j = 0; j < temp.length; j++) {
                    let tempIndex = target.indexOf(temp[j])
                    if (curr < tempIndex) {
                        temp[j] = arr[i]
                        break
                    }
                    if (tempIndex === curr) {
                        break
                    }
                }
            }  
        }
    }
    return target.length - temp.length
};
